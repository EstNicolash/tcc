	\section{Estruturas de Kripke}
	Uma estrutura de Kripke (ou um sistema de transição) é uma tupla $(S, R, I,AP, L)$ onde:
	\begin{itemize}
		\item $S$ é um conjunto finito de \textbf{estados} (ou mundos possíveis).
		\item $R \subseteq S \times S$ é uma \textbf{relação de acessibilidade} total (para cada $s \in S$, existe $s' \in S$ tal que $(s, s')\in R$).
		\item $I \subseteq S$ é o conjunto de \textbf{estados iniciais}.
		\item $AP$ é um conjunto finito de \textbf{proposições atômicas}.
		\item $L: S \to 2^{AP}$ é uma \textbf{função de rotulação} que associa a cada estado o conjunto de proposições que nele são verdadeiras.
	\end{itemize}
	
	\section{Switching Functions}
	A verificação simbólica opera sobre conjuntos de estados representados por funções boolianas.
	
	\subsection*{Variáveis e avaliações:}
	Seja um conjunto de variáveis boolianas $Var = \{x_1,x_2,x_3,...,x_n\}$. Uma \textbf{avaliação} (do conjunto) é uma atribuição de valores booleanos (0 ou 1) às variáveis (do conjunto), ou seja, uma avaliação é um função $\eta: Var\mapsto \{0,1\}$. $Eval(Var)$ é o conjunto de \textbf{todas as avaliações (do conjunto de variáveis)}.
	
	\subsection*{Switching Functions}
	Uma \textbf{switching function} é uma função $f: Eval(Var) \mapsto \{0,1\}$. No contexto de verificação, a \textbf{função característica} $\chi_T$ para representar um subconjunto de estados $T \subseteq S$:
	\begin{equation}
		\chi_T(s) = 1 \iff s \in T
	\end{equation}
	Desta forma, operações sobre conjuntos (união, interseção, etc) são realizadas por operações lógicas ($\lor$,$\land$, etc.) sobre \textbf{switching functions}.  
	
	\subsection*{Cofatores}
	Um cofator de uma função $f$ é como fixasse uma das variáveis com um valor booliano. Seja $f:Eval(y,x_1,x_2,...,x_n)\to\{0,1\}$,  cofator positivo é $f|_{x=1}:Eval(1,x_1,x_2,...,x_n)\to\{0,1\}$ e o cofator negativo é $f|_{x=0}:Eval(0,x_1,x_2,...,x_n)\to\{0,1\}$. Caso $f|_{x=1}\neq f|_{x=0}$, $x$ é considerado uma \textbf{variável essencial}.
	
	\subsection*{Quantificadores}
	\begin{equation}
		\exists x.f = f|_{x=0}\lor f |_{x=1}
	\end{equation}
	\begin{equation}
		\forall x.f = f|_{x=0}\land f |_{x=1}\\
	\end{equation}
	
	\subsection*{Operação de renomeação}
	Exemplo: seja $f = x \land(\neg x\lor y)$. Renomeando $x$ para $x'$ temos:
	$f\{x\gets x'\} = x' \land(\neg x'\lor y)$.
	
	\section{Binary Decision Trees}
	\section{Binary Decision Diagram}
	
	Seja um conjunto de variáveis $Var =\{x_1,x_2,...,x_n\}$. Um \textit{Binary Decision Diagram} é uma tupla $(V,V_I,V_T,succ_0,succ_1,var,val,v_0)$ em que temos:
	\begin{itemize}
		\item Um conjunto finito de nós $V$.
		\item Os conjuntos $V_I\subset V$ e $V_T\subset V$ tal que $V_I$ são os nós internos, $V_T$ são nós terminais e $V_I\cap V_T=\emptyset$.
		\item As funções de sucessores $succ_0: V_I\to V$ e  $succ_1: V_I\to V$.
		\item A função de atribuição de nós internos a variáveis $var: V_I\to Var$.
		\item A função de atribuição de valores boolianos a nós terminais $val: V_T\to \{0,1\}$.
		\item Um nó raiz $v_0$ em que $\neg\exists x(succ_1(x)=v_0\lor succ_0(x)=v_0)$.
	\end{itemize}

	\section{Ordered Binary Decision Diagram (OBDD)}
	Um $\mathcal{P}$-OBDD é um BDD $\mathcal{B}$ que respeita um ordenamento de variáveis $(x_1,x_2,...,x_n)$ em que $x_i<x_{i+1}$ para qualquer valor de $i\in{\{1,...,n\}}$. A função de sucessores deve ser consistente com a relação de ordem, ou seja, se $succ(v_i) = v_j$, então, $var(v_i)<var(v_j)$. \\

	\section{Reduced Ordered Binary Decision Diagram (ROBDD)}
	Um ROBDD é um $\mathcal{P}$-OBDD em que $\forall v,w\in V(v\neq w\to f_v \neq f_w)$. $f_v$ é a função induzida de um BDD tomando o nó $v$ como raiz. \\
	
	\subsection*{Exemplo: Função e BDD induzido}
	Seja o BDD para $f = x \land y$. Cada nó $v$ induz uma função $f_v$ baseada em sua posição na estrutura:
	
	\begin{center}
		\begin{tikzpicture}[node distance=1.5cm, v/.style={circle,draw,minimum size=8mm}, t/.style={rectangle,draw,minimum size=6mm}, >=stealth, thick]
			\node[v] (vx) {$x$};
			\node[v, below right=of vx] (vy) {$y$};
			\node[t, below left=of vx] (t0) {0};
			\node[t, below=of vy] (t1) {1};
			
			\draw[->, dashed] (vx) -- (t0);
			\draw[->] (vx) -- (vy);
			\draw[->, dashed] (vy) -- (t0);
			\draw[->] (vy) -- (t1);
			
			% Notas explicativas
			\node[right=0.5cm of vx, blue] {$f_{vx} = x \land y$};
			\node[right=0.5cm of vy, blue] {$f_{vy} = y$};
			\node[left=0.5cm of t0, blue] {$f_{t0} = 0$};
		\end{tikzpicture}
	\end{center}
	Note que a função induzida pela raiz é a própria função alvo, enquanto o nó $y$ induz a função correspondente ao seu valor de verdade, já que $x$ foi avaliado como 1.
	
	\subsection*{Regras de redução:}
	ROBDD é a forma canônica de um OBDD, ou seja, todo OBDD pode ser transformado num ROBDD por meio de uma transoformação. Dado um $\mathcal{B}=(V,V_I,V_T,succ_0,succ_1,var,val,v_0)$, as regras de redução são:
	\begin{itemize}
		\item \textbf{Regra de eliminação:} Se $v\in V_I$ com $succ_0(v)= succ_1(v)=w$, então, $v$ deve ser removido e, todo $u$, tal que $succ_1(u) = v$, será redirecionado para $w$,  $succ_1(u) = w$. O mesmo se aplica para o caso em que $succ_0(u)=v$.
		\item \textbf{Regra de isomorfismo:} Se $v,w\in V$, $v\neq w$ e $v,w\in V_T$ atribuídos a um mesmo valor ($val(v)=val(w)$) ou $v,w\in V_I$ com $(var(v),succ_1(v),succ_0(v))=(var(w),succ_1(w),succ_0(w))$, então remove $v$ e redireciona os nós $u$ em que $succ_1(u) = v$ para $w$ ($succ_1(u) = w$). O mesmo se aplica para $succ_0(u)=v$.
	\end{itemize}

	\section{Shared Ordered Binary Decision Diagram (SOBDD)}
	Cada função possui algum ROBDD para ser representada, porém, manter diversos ROBDDs pode ser custoso, pois há redundâncias e e duplicação de funções. Para evitar isto, há o SOBDDS, um OBDD com diversas raízes (um conjunto de raízes ao invés de um único $v_0$). Cada função é representada por uma raiz e as diferentes funções compartilham nós de funções isomórficas.
	
	\subsection*{Exemplo: Shared OBDD (SOBDD)}
	Neste exemplo, as funções $f = x \land y$ e $g = z \lor (x \land y)$ são representadas simultaneamente. A subexpressão $(x \land y)$ é calculada e armazenada apenas uma vez, sendo compartilhada por ambas as raízes.
	
	\begin{center}
		\begin{tikzpicture}[
			node distance=1.5cm and 2cm,
			v/.style={circle, draw, minimum size=8mm, thick},
			t/.style={rectangle, draw, minimum size=6mm, thick},
			root/.style={draw, stealth-, thick, red!70!black},
			>=stealth,
			thick
			]
			
			% Nós da estrutura
			\node[v] (z) {$z$};
			\node[v] (x) [below left=of z, xshift=1cm] {$x$};
			\node[v] (y) [below right=of x, xshift=-1cm] {$y$};
			
			% Terminais únicos
			\node[t] (t0) [below left=1.5cm of y] {0};
			\node[t] (t1) [below right=1.5cm of y] {1};
			
			% Definição das Raízes (Entry Points)
			\draw[root] (z.north) -- ++(0,0.7) node[above] {Raiz de $g$};
			\draw[root] (x.north west) -- ++(-0.5,0.5) node[left] {Raiz de $f$};
			
			% Arestas de Z (Função g)
			\draw[->] (z) -- (t1) node[midway, right, scale=0.8] {1}; % Se z=1, g=1

	\chapter{Mapeamento de Modelos SSMV para BDDs}
	Para converter um modelo SSMV para a representação simbólica, seguimos os seguintes passos:
	
	\section{Codificação de Variáveis}
	Variáveis de domínios finitos são mapeadas para vetores de bits booleanos de acordo com o tamanho do domínio do tipo de dado. Seja $len(x)$ o domínio de uma variável do ssmv (inteiro, booliana ou enumeração). A quantidade de variáveis (boolianas do BDD) necessárias para representar cada variável do ssmv é igual a $\lceil \log_2(len(x))\rceil$. 
	
	\subsection*{Exemplo}
	Caso $n: 0..7$, teremos $len(n) = 8$, assim $\lceil \log_2(len(n))\rceil = \lceil \log_2(8)\rceil = 3$, ou seja, precisaremos de 3 variáveis para representar a variável $n$ do ssmv.
	
	\section{Codificação de expressões boolianas}
	Para cada expressão booliana em ssmv, basta compilar as subexpressões para um BDD e realizar a respectiva operação booliana. Seja $f_a$ a função representada pela compilação da expressão ssmv $a$ para um BDD.
	\begin{itemize}
		\item \textbf{a \& b}: $f_{\text{a\&b}} = f_a\land f_b$.
		\item \textbf{a | b}: $f_{\text{a|b}} = f_a\lor f_b$.
		\item \textbf{!a}: $f_{\text{!a}} = \neg f_a$.
		\item \textbf{a->b}: $f_{\text{a->b}} = f_a\to f_b$.
	\end{itemize}
	
	\section{Codificação de expressões relacionais}
	Seja $n = len(a) = len(b)$, $x_i$ são as variáveis codificadas de \textbf{a} e  $y_i$ são as variáveis codificadas de \textbf{b}.
	\begin{itemize}
		\item \textbf{a = b}: $f_{\text{a=b}} =  \bigwedge_{i=1}^n (x_i \leftrightarrow y_i)$
		\item \textbf{a > b}: $f_{\text{a>b}} = \bigvee_{i=1}^n \left( (x_i \land \neg y_i) \land \bigwedge_{j=i+1}^n (x_j \leftrightarrow y_j) \right)$
		\item \textbf{a>=b}: $f_{\text{a>=b}} = f_{\text{a=b}}\lor f_{\text{a>b}} $
		\item \textbf{a<=b}: $f_{\text{a<=b}} = \neg f_{\text{a>b}} $
		\item \textbf{a<b}: $f_{\text{a<b}} = \neg f_{\text{a>=b}} $
	\end{itemize}
	
	\section{Codificação de Operações Aritméticas}
	A aritmética de vetores de bits é implementada simulando circuitos combinatórios através de operações lógicas em BDDs. 
	
	\subsection*{Adição (Ripple Carry Adder)}
	A soma de dois vetores de bits $\textbf{a}$ e $\textbf{b}$ de tamanho $n$ é realizada bit a bit, propagando um bit de transporte (\textit{carry}). O bit de transporte inicial é definido como $c_0 = 0$. Para cada bit $i \in \{1, \dots, n\}$, o bit de soma $s_i$ e o próximo \textit{carry} $c_{i}$ são calculados como:
	\begin{itemize}
		\item $s_i = x_i \oplus y_i \oplus c_{i-1}$
		\item $c_i = (x_i \land y_i) \lor ((x_i \oplus y_i) \land c_{i-1})$
	\end{itemize}
	
	\subsection*{Subtração (Complemento de Dois)}
	A subtração $\textbf{a} - \textbf{b}$ é convertida em uma operação de soma utilizando a representação de complemento de dois. Matematicamente, a operação é equivalente a $\textbf{a} + \neg \textbf{b} + 1$. 
	\begin{itemize}
		\item O \textit{carry}inicial é definido como $c_0 = 1$, representando o $+1$ da fórmula.
		\item $sub_i = x_i \oplus \neg y_i \oplus c_{i-1}$.
	\end{itemize}
	
	\section{Relações de atribuição}
	\subsection*{Conjuntos}
	Seja um conjunto $\{e_1,e_2,...,e_k\}$, $f_{x'}=e_i$ a função que atribui $e_i$ a variável futura $x'$.
	\begin{equation}
		f_{enum} = \bigvee_{i=1}^k f_{x'=e_i}
	\end{equation}
	
	\subsection*{Condicionais}
	A expressão \textbf{\textit{cond}} do ssmv é basciamente um \textit{if-then-else} encadeado, assim, pode ser representado pela operação $ITE(x,y,z)$ de BDDs.
	\begin{equation}
		f_{case} = ITE(c_1, f_{x'=r_1}, ITE(c_2, f_{x'=r_2}, \dots ITE(c_n, f_{x'=r_n}, \text{false}) \dots ))
	\end{equation}
	
	\section{Relação de Transição (\texorpdfstring{$\Delta$}{Delta})}
	A relação de transição é codificada como uma fórmula booliana relacionando variáveis do presente ($x$) e do futuro ($x'$).
	\begin{equation}
		\Delta(s, s') = \bigwedge_{x \in Var} \mathcal{R}_x(s, s')
	\end{equation}
	
	Onde $Rv(s,s')$ é o BDD construído para a variável $x$, integrando as regras de:
	\begin{itemize}
		\item \textbf{Atribuição Simples:} $R_x=(x'\leftrightarrow f_x(s))$
		\item \textbf{Conjuntos:} $R_x= \bigvee_{i=1}^k(x'\leftrightarrow e_i)$
		\item \textbf{Condicionais (Case):} $R_v = ITE(c_1,R_{x1},ITE(c_2,R_{x2},…))$
	\end{itemize}
